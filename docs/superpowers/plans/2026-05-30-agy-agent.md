# Agy Agent Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `agy` (Antigravity CLI) as a supported agent in cc-connect, following the same plugin pattern as `gemini`.

**Architecture:** agy uses a per-message subprocess spawn model identical to Gemini CLI: each `Send()` launches `agy --dangerously-skip-permissions -p -` (reading prompt from stdin) and optionally `--continue` or `--conversation <id>` for session continuity. Output is plain text (not stream-json), so we read stdout as a plain text result. Permission prompts are not applicable since `--dangerously-skip-permissions` is always passed.

**Tech Stack:** Go, `os/exec`, `bufio`, `core.Event` channel, identical patterns to `agent/gemini/`.

```go
package agy

import (
	"context"
	"fmt"
	"log/slog"
	"os/exec"
	"strings"
	"sync"
	"time"

	"github.com/chenhg5/cc-connect/core"
)

func init() {
	core.RegisterAgent("agy", New)
}

// Agent drives the Antigravity CLI (agy) in headless mode using -p (--print).
// Each Send() spawns a new subprocess. Session continuity via --conversation.
type Agent struct {
	workDir string
	cmd     string
	timeout time.Duration
	mu      sync.RWMutex
}

func New(opts map[string]any) (core.Agent, error) {
	workDir, _ := opts["work_dir"].(string)
	if workDir == "" {
		workDir = "."
	}
	cmd, _ := opts["cmd"].(string)
	if cmd == "" {
		cmd = "agy"
	}

	var timeoutMins int64
	switch v := opts["timeout_mins"].(type) {
	case int64:
		timeoutMins = v
	case int:
		timeoutMins = int64(v)
	case float64:
		timeoutMins = int64(v)
	default:
		if v != nil {
			slog.Debug("agy: timeout_mins has unexpected type", "type", fmt.Sprintf("%T", v))
		}
	}
	var timeout time.Duration
	if timeoutMins > 0 {
		timeout = time.Duration(timeoutMins) * time.Minute
	}

	if _, err := exec.LookPath(cmd); err != nil {
		return nil, fmt.Errorf("agy: %q not found in PATH; ensure agy is installed", cmd)
	}

	return &Agent{workDir: workDir, cmd: cmd, timeout: timeout}, nil
}

// NormalizeModeExported is exported only for unit tests.
func NormalizeModeExported(raw string) string { return normalizeMode(raw) }

func normalizeMode(raw string) string {
	switch strings.ToLower(strings.TrimSpace(raw)) {
	case "yolo", "auto", "force":
		return "yolo"
	default:
		return "default"
	}
}

func (a *Agent) Name() string            { return "agy" }
func (a *Agent) CLIBinaryName() string   { return a.cmd }
func (a *Agent) CLIDisplayName() string  { return "Agy" }

func (a *Agent) SetWorkDir(dir string) {
	a.mu.Lock()
	defer a.mu.Unlock()
	a.workDir = dir
	slog.Info("agy: work_dir changed", "work_dir", dir)
}

func (a *Agent) GetWorkDir() string {
	a.mu.Lock()
	defer a.mu.Unlock()
	return a.workDir
}

func (a *Agent) StartSession(ctx context.Context, sessionID string) (core.AgentSession, error) {
	a.mu.Lock()
	cmd, workDir, timeout := a.cmd, a.workDir, a.timeout
	a.mu.Unlock()
	return newAgySession(ctx, cmd, workDir, sessionID, timeout)
}

// ListSessions returns nil — agy stores conversations as protobuf blobs
// (~/.gemini/antigravity-cli/conversations/*.pb) without a per-workdir index.
func (a *Agent) ListSessions(_ context.Context) ([]core.AgentSessionInfo, error) {
	return nil, nil
}

func (a *Agent) DeleteSession(_ context.Context, _ string) error {
	return fmt.Errorf("agy: session deletion is not supported")
}

func (a *Agent) Stop() error { return nil }
```

```go
	defer cancel()

	sess, err := newAgySessionWithArgs(ctx, "/bin/sh", []string{"-c", "cat"}, "/tmp", "", 0)

		t.Fatalf("newAgySession: %v", err)

	if err := sess.Send("hello world", nil, nil); err != nil {
		t.Fatalf("Send: %v", err)

	var texts []string
	gotDone := false
	timer := time.After(5 * time.Second)
	for !gotDone {
		select {
		case evt, ok := <-sess.Events():
			if !ok {
				t.Fatal("events channel closed unexpectedly")

			case core.EventText:

			t.Fatal("timed out waiting for events")
		}

	if !strings.Contains(combined, "hello world") {
		t.Errorf("expected 'hello world' in response, got: %q", combined)
```

```go
package agy

import (
	"bufio"
	"bytes"
	"context"
	"fmt"
	"io"
	"log/slog"
	"os"
	"os/exec"
	"strings"
	"sync"
	"sync/atomic"
	"time"

	"github.com/chenhg5/cc-connect/core"
)

// agySession manages multi-turn conversations with the Antigravity CLI.
//
// Each Send() launches: agy --dangerously-skip-permissions [--conversation <id>] -p -
// with the prompt on stdin. Output is plain text; entire stdout is emitted as
// a single EventText followed by EventDone.
type agySession struct {
	cmd      string
	cmdArgs  []string // override args for testing (nil in production)
	workDir  string
	timeout  time.Duration

	events         chan core.Event
	conversationID atomic.Value // stores string

	ctx    context.Context
	cancel context.CancelFunc
	wg     sync.WaitGroup
	alive  atomic.Bool

	cmdMu sync.Mutex
	cmds  map[*exec.Cmd]struct{}
}

func newAgySession(ctx context.Context, cmd, workDir, resumeID string, timeout time.Duration) (*agySession, error) {
	return newAgySessionWithArgs(ctx, cmd, nil, workDir, resumeID, timeout)
}

// newAgySessionWithArgs is the internal constructor; cmdArgs overrides the
// subprocess args (used in tests to inject a fake command).
func newAgySessionWithArgs(ctx context.Context, cmd string, cmdArgs []string, workDir, resumeID string, timeout time.Duration) (*agySession, error) {
	sessCtx, cancel := context.WithCancel(ctx)
	s := &agySession{
		cmd:     cmd,
		cmdArgs: cmdArgs,
		workDir: workDir,
		timeout: timeout,
		events:  make(chan core.Event, 64),
		ctx:     sessCtx,
		cancel:  cancel,
		cmds:    make(map[*exec.Cmd]struct{}),
	}
	s.alive.Store(true)
	if resumeID != "" && resumeID != core.ContinueSession {
		s.conversationID.Store(resumeID)
	}
	return s, nil
}

func (s *agySession) Events() <-chan core.Event { return s.events }

func (s *agySession) CurrentSessionID() string {
	v, _ := s.conversationID.Load().(string)
	return v
}

// RespondPermission is a no-op: agy runs with --dangerously-skip-permissions.
func (s *agySession) RespondPermission(_ string, _ core.PermissionResult) error {
	return nil
}

func (s *agySession) Send(prompt string, images []core.ImageAttachment, files []core.FileAttachment) error {
	if !s.alive.Load() {
		return fmt.Errorf("agy: session closed")
	}

	var execCmd *exec.Cmd
	convID := s.CurrentSessionID()

	var ctx context.Context
	var cancel context.CancelFunc
	if s.timeout > 0 {
		ctx, cancel = context.WithTimeout(s.ctx, s.timeout)
	} else {
		ctx, cancel = context.WithCancel(s.ctx)
	}

	started := false
	defer func() {
		if !started {
			cancel()
		}
	}()

	if len(s.cmdArgs) > 0 {
		// Test mode: run the injected command directly.
		execCmd = exec.CommandContext(ctx, s.cmd, s.cmdArgs...)
	} else {
		// Production mode: build the real agy invocation.
		args := []string{"--dangerously-skip-permissions"}
		if convID != "" {
			args = append(args, "--conversation", convID)
		} else {
			args = append(args, "--continue")
		}
		args = append(args, "-p", "-")
		slog.Debug("agySession: launching", "conversation_id", convID, "args", core.RedactArgs(args))
		execCmd = exec.CommandContext(ctx, s.cmd, args...)
	}

	execCmd.WaitDelay = 1 * time.Second
	execCmd.Dir = s.workDir
	execCmd.Env = os.Environ()

	// Build the full prompt, appending attachment references.
	fullPrompt := prompt
	if len(images) > 0 || len(files) > 0 {
		var refs []string
		for _, img := range images {
			refs = append(refs, "[image attachment: "+img.MimeType+"]")
		}
		for _, f := range files {
			refs = append(refs, "[file attachment: "+f.FileName+"]")
		}
		fullPrompt += "\n\n" + strings.Join(refs, "\n")
	}
	execCmd.Stdin = strings.NewReader(fullPrompt)

	stdout, err := execCmd.StdoutPipe()
	if err != nil {
		return fmt.Errorf("agySession: stdout pipe: %w", err)
	}
	var stderrBuf bytes.Buffer
	execCmd.Stderr = &stderrBuf

	if err := execCmd.Start(); err != nil {
		return fmt.Errorf("agySession: start: %w", err)
	}

	s.cmdMu.Lock()
	s.cmds[execCmd] = struct{}{}
	s.cmdMu.Unlock()

	started = true
	s.wg.Add(1)
	go func() {
		defer cancel()
		defer s.wg.Done()
		s.readLoop(ctx, execCmd, stdout, &stderrBuf)
	}()
	return nil
}

func (s *agySession) readLoop(ctx context.Context, cmd *exec.Cmd, stdout io.ReadCloser, stderrBuf *bytes.Buffer) {
	defer func() {
		s.cmdMu.Lock()
		delete(s.cmds, cmd)
		s.cmdMu.Unlock()
	}()

	go func() {
		<-ctx.Done()
		stdout.Close()
	}()

	var sb strings.Builder
	scanner := bufio.NewScanner(stdout)
	scanner.Buffer(make([]byte, 0, 64*1024), 4*1024*1024)
	for scanner.Scan() {
		sb.WriteString(scanner.Text())
		sb.WriteByte('\n')
	}

	if err := cmd.Wait(); err != nil {
		stderr := strings.TrimSpace(stderrBuf.String())
		if stderr != "" {
			slog.Error("agySession: process failed", "error", err, "stderr", stderr)
			s.emitEvent(core.Event{Type: core.EventError, Error: fmt.Errorf("agy: %s", stderr)})
			s.emitEvent(core.Event{Type: core.EventDone})
			return
		}
		if ctx.Err() != nil {
			return // cancelled intentionally
		}
	}

	text := strings.TrimRight(sb.String(), "\n")
	if text == "" {
		s.emitEvent(core.Event{Type: core.EventDone})
		return
	}

	// Detect authentication error output.
	if strings.Contains(text, "Authentication required") ||
		strings.Contains(text, "authentication interrupted") {
		slog.Error("agySession: authentication required", "output", text)
		s.emitEvent(core.Event{
			Type:  core.EventError,
			Error: fmt.Errorf("agy: authentication required — run agy interactively to log in first"),
		})
		s.emitEvent(core.Event{Type: core.EventDone})
		return
	}

	s.emitEvent(core.Event{Type: core.EventText, Content: text})
	s.emitEvent(core.Event{Type: core.EventDone})
}

func (s *agySession) emitEvent(evt core.Event) {
	select {
	case s.events <- evt:
	case <-s.ctx.Done():
	}
}

func (s *agySession) Close() error {
	if !s.alive.CompareAndSwap(true, false) {
		return nil
	}
	s.cancel()
	s.cmdMu.Lock()
	for cmd := range s.cmds {
		if cmd.Process != nil {
			_ = cmd.Process.Kill()
		}
	}
	s.cmdMu.Unlock()
	s.wg.Wait()
	close(s.events)
	return nil
}
```
